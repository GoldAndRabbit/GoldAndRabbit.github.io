е·®еҲҶж•°з»„: й«ҳж•Ҳең°е®һзҺ°еҢәй—ҙеўһеҮҸ

1.жғідёҖдёӘеңәжҷҜ, йңҖиҰҒйў‘з№Ғең°з»ҷж•°з»„зҡ„жҹҗдәӣдёҚеҗҢзҡ„еҢәй—ҙйў‘з№Ғзҡ„еўһеҠ жҲ–иҖ…еҮҸе°‘, дҫӢз»ҷз»ҷе®ҡдёҖдёӘж•°з»„ nums, з„¶еҗҺз»ҷ nums[2..6] йғҪ +1, з„¶еҗҺеҜ№ nums[3..0]йғҪ -3, з„¶еҗҺеҜ№ nums[0..4] йғҪ+2, жңҙзҙ зҡ„жғіжі•жҳҜд»Қз„¶жҳҜеӨҡж¬Ў O(n) жү«жҸҸ
2.жғідёҖдёӢиҝҷйҮҢжңүжІЎжңүдјҳеҢ–зҡ„з©әй—ҙ
3.е·®еҲҶж•°з»„зҡ„дҪҝз”ЁеңәжҷҜжҳҜйў‘з№Ғең°еҜ№ж•°з»„зҡ„жҹҗдёӘеҢәй—ҙиҝӣиЎҢе…ғзҙ иҝӣиЎҢеўһеҮҸ, жһ„йҖ е·®еҲҶж•°з»„ diff[i] = nums[i] - nums[i-1]
еҺҹе§Ӣж•°з»„ nums = [8,2,6,3,1]
е·®еҲҶж•°з»„ diff = [8,-6,4,-3,2]
4.еҲ©з”Ёе·®еҲҶж•°з»„д№ҹеҸҜд»Ҙжһ„йҖ еҮәжқҘеҺҹе§Ӣж•°з»„, еҜ№дәҺ nums[i] = diff[i] + nums[i-1]

жһ„йҖ е·®еҲҶж•°з»„ 

// жһ„йҖ е·®еҲҶж•°з»„  
diff[0] = nums[0]  
diff[1] = nums[1] - nums[0]
diff[2] = nums[2] - nums[1]

иҝҳеҺҹж•°з»„

// иҝҳеҺҹж•°з»„
nums[0] = diff[0]
nums[1] = diff[1] + nums[0] = diff[1] + diff[2]
nums[2] = diff[2] + nums[1] = diff[2] + diff[1] + diff[0]

vector<int> res(n);
res[0] = diff[0];
for (int i = 1; i < diff.size(); ++i) {
  res[i] = res[i-1] + diff[i];
}

5.жңүдәҶе·®еҲҶж•°з»„, еҰӮдҪ•е®һзҺ°ж•°з»„еҢәй—ҙеўһеҮҸ? еҰӮжһңжҲ‘д»¬зӣёеҜ№ nums[i..j] еҢәй—ҙе…ЁйғЁ+3, жҲ‘д»¬йңҖиҰҒ diff[i] += 3, з„¶еҗҺ diff[j+1] -= 3, з„¶еҗҺйҮҚж–°ж“ҚдҪңиҝҳеҺҹж•°з»„иҝҮзЁӢ, жғідёҖдёӢжҖҺд№Ҳе®һзҺ°зҡ„, еҒҮи®ҫжҲ‘д»¬еҜ№еҺҹе§Ӣзҡ„ж•°з»„ index[1..3] йғҪ +3
еҺҹе§Ӣж•°з»„ nums [8,2,6,3,1]
е·®еҲҶж•°з»„ diff [8,-6,4,-3,2]

жӣҙж–°еҗҺзҡ„еҺҹе§Ӣж•°з»„nums [8,5,9,6,1] еҜ№еҺҹе§Ӣзҡ„ж•°з»„ index[1..3] йғҪ +3
жӣҙж–°еҗҺзҡ„е·®еҲҶж•°з»„diff [8,-3,4,-3,-5] е’ҢеҺҹе§Ӣе·®еҲҶж•°з»„зҡ„еҢәеҲ«еңЁдәҺ diff[i] += 3 && diff[j+1] -= 3

жҖқиҖғе·®еҲҶж•°з»„ diff[i] += 3 иҫҫеҲ°дәҶд»Җд№Ҳж•Ҳжһң, зӣёеҪ“дәҺеҜ№ nums[i..] еҗҺйқўзҡ„жүҖжңүе…ғзҙ йғҪ+3, жҚўеҸҘиҜқиҜҙ, еҜ№еҪ“еүҚзҡ„е…ғзҙ жҜ”еүҚдёҖдёӘе…ғзҙ еўһеҠ дәҶ 3, йӮЈд№ҲеҗҺйқўзҡ„е…ғзҙ д»Қз„¶жҜ”еүҚдёҖдёӘе…ғзҙ еӨ§ diff[i+1], diff[i+2], иҝҷдёӘеҖјйғҪиў«иў«еҠЁзҡ„еҠ дёҠдәҶ3
еҗҢзҗҶ diff[j+1] -= 3, еҜ№д»Һ nums[j+1..] еҗҺйқўзҡ„е…ғзҙ жүҖжңүйғҪ-3, иҝӣиЎҢ +3 е’Ң -3 ж“ҚдҪңд№ӢеҗҺйғҪжІЎжңүеҸҳ
еӣ жӯӨ, еҪ“жҲ‘д»¬зӣёеҜ№ nums[i..j] йғҪиҝӣиЎҢ+xзҡ„ж“ҚдҪң, еҸӘйңҖиҰҒеҜ№ diff[i]+=x, дё”еҜ№ diff[j+3]-=x, з„¶еҗҺеҲ©з”Ёе·®еҲҶж•°з»„еҜ№еҺҹе§Ӣзҡ„ж“ҚдҪңиҝӣиЎҢиҝҳеҺҹж“ҚдҪң

1109.иҲӘзҸӯйў„е®ҡз»ҹи®Ў

иҝҷйҮҢжңү n дёӘиҲӘзҸӯпјҢе®ғд»¬еҲҶеҲ«д»Һ 1 еҲ° n иҝӣиЎҢзј–еҸ·гҖӮ
жңүдёҖд»ҪиҲӘзҸӯйў„и®ўиЎЁ bookings пјҢиЎЁдёӯз¬¬ i жқЎйў„и®ўи®°еҪ• bookings[i] = [firsti, lasti, seatsi] ж„Ҹе‘ізқҖеңЁд»Һ firsti еҲ° lasti пјҲеҢ…еҗ« firsti е’Ң lasti пјүзҡ„ жҜҸдёӘиҲӘзҸӯ дёҠйў„и®ўдәҶ seatsi дёӘеә§дҪҚгҖӮ
иҜ·дҪ иҝ”еӣһдёҖдёӘй•ҝеәҰдёә n зҡ„ж•°з»„ answerпјҢйҮҢйқўзҡ„е…ғзҙ жҳҜжҜҸдёӘиҲӘзҸӯйў„е®ҡзҡ„еә§дҪҚжҖ»ж•°гҖӮ

йў„и®ўи®°еҪ• 2 пјҡ 20 20

иҫ“е…Ҙпјҡbookings = [[1,2,10],[2,2,15]], n = 2

и§ЈйҮҠпјҡ

йў„и®ўи®°еҪ• 2 пјҡ 15

еҲҶжһҗ:
1.еә§дҪҚж•°е°ұжҳҜеҢәй—ҙеҶ…зҡ„е’Ң, з”Ёе·®еҲҶж•°з»„иҫ…еҠ©з»ҹи®Ў 

class Solution {
 private:
  vector<int> nums;
  vector<int> diff;

 public:
  void buildDiff() {
    diff[0] = nums[0];
    for (int i = 1; i < nums.size(); ++i) {
      diff[i] = nums[i] - nums[i-1];
    }
    return;
  }

  vector<int> recoverNums() {
    vector<int> res(nums.size());
    res[0] = diff[0];
    for (int i = 1; i < diff.size(); ++i) {
      res[i] = res[i-1] + diff[i];
    }
    return res;
  }

  void increment(int l, int r, int val) {
    diff[l] += val;
    if (r+1 < diff.size()) {
        diff[r+1] -= val;
    }
    return;
  }

  vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
    nums.resize(n, 0);
    diff.resize(n);
    buildDiff();
    for (int i = 0; i < bookings.size(); ++i) {
      int l = bookings[i][0]-1;
      int r = bookings[i][1]-1;
      int v = bookings[i][2];
      increment(l, r, v);
    }
    return recoverNums();
  }
};

class Solution {
 private:
  vector<int> nums;
  vector<int> diff;
 public:
  void buildDiff() {
    diff[0] = nums[0];
    for (int i = 1; i < nums.size(); ++i) {
      diff[i] = nums[i] - nums[i-1];
    }
  }
  void increment(int i, int j, int val) {
    diff[i] += val;
    if (j+1 < diff.size()) {
      diff[j+1] -= val;
    }
  }
  vector<int> recover() {
    vector<int> ret(nums.size());
    ret[0] = diff[0];
    for (int i = 1; i < diff.size(); ++i) {
      ret[i] = ret[i-1] + diff[i];
    }
    return ret;
  }
  bool carPooling(vector<vector<int>>& trips, int capacity) {
    nums.resize(1000+1, 0);
    diff.resize(1000+1);
    for (vector<int> trip : trips) {
      int i = trip[1];
      int j = trip[2]-1;
      int val = trip[0];
      increment(i, j, val);
    }
    vector<int> ret = recover();
    int max = *max_element(ret.begin(), ret.end());
    return max <= capacity;
  }
};
